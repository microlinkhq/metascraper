smartlink-core

Get metadata from HTML.

Install

$ npm install smartlink-core --save

Usage

const smartlink = require('smartlink-core')
const get = require('simple-get')

get.concat('http://example.com', function (err, res, html) {
  if (err) throw err

  const output = smartlink(html)
  console.log(output)
})
